IME_irgun_up_2 = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = {
					var = IME_irgun_strength
					value = 0
					compare = greater_than
				}
				NOT = {
					has_country_flag = IME_irgun_wiped_out
				}
			}
			add_to_variable = {
				var = IME_irgun_strength
				value = 2
			}
			clamp_variable = {
				var = IME_irgun_strength
				max = 40
				min = 0
			}
		}
	}
	custom_effect_tooltip = IME_empower_irgun
}
