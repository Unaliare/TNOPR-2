# This specifically overrides the rule check as to whether or not you can guarantee other ideologies

DIPLOMACY_CALL_ALLY_ENABLE_TRIGGER = {
	if = {
		limit = {
			ROOT = {
				AND = {
					OR = {
						tag = JAP
						tag = MEN
						tag = MAN
						tag = CHI
					}
					OR = {
						has_war_with = PRC
						has_war_with = MON
					}
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = DIPLO_MON_WAR_CALL
			always = no
		}
	}
	else = {
		always = yes
	}
}
